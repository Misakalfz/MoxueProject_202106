/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public int[] reversePrint(ListNode head) {
        if (head == null){;
            return new int[0];
        }
        ListNode cur = head, pre = null;
        int size = 0;
        while (cur != null){
            ListNode temp = cur.next;
            cur.next = pre;
            pre = cur;
            cur = temp;
            size ++;
        }
        int[] res = new int[size];

        for (int i = 0; i < size; i++){
            res[i] = pre.val;
            pre = pre.next;
        }
        return res;
    }
}
