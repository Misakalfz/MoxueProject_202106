class Solution {
    public int[] twoSum(int[] numbers, int target) {
        if (numbers.length < 2){
            return null;
        }
        int left = 0, right = numbers.length - 1;
        while (left <= right){
            if (numbers[left] + numbers[right] < target){
                left++;
            }
            if (numbers[left] + numbers[right] > target){
                right--;
            }
            else{
                int[] res = {left + 1, right + 1};
                return res;
            }
        }
        return null;
    }
}
