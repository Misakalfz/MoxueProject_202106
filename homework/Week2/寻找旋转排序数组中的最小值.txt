class Solution {
    public int findMin(int[] nums) {
        int left = 0;
        int right = nums.length - 1;
        while (left < right){
            int pivot = (right - left) / 2 + left;
            if (nums[pivot] < nums[right]){
                right = pivot;
            }
            else {
                left = pivot + 1;
            }
        }
        return nums[left];

    }
}
